KBC App: 0 to Infinity Lesson

Build a KBC style quiz app using HTML, CSS, JavaScript, JSON data, radio buttons, timer, rewards and speaking voice.

What We Will Build

We will build a simple KBC style quiz app from zero. The app starts from a basic HTML page and slowly grows into a complete interactive quiz project.

  • Start screen
  • Game screen
  • 15 JSON questions
  • Radio button options
  • Hidden default radio
  • 30 seconds timer
  • Answer checking
  • Rewards JSON
  • Speaking question
  • KBC style CSS

Stage 4: Add 30 Seconds Counter

We will use setInterval() to reduce the counter every second. 

let counter = 30;
let timerId = null;

function startTimer() {
    clearInterval(timerId);

    counter = 30;
    lbltimer.textContent = counter;

    timerId = setInterval(function () {
        counter--;
        lbltimer.textContent = counter;

        if (counter === 0) {
            clearInterval(timerId);
            alert("Time up!");
        }
    }, 1000);
}
Important: 1000 milliseconds means 1 second.

Stage 11: Create loadNextQuestion()

function loadNextQuestion() {
    questionIndex++;

    if (questionIndex >= questions.length) {
        alert("Quiz finished!");
        return;
    }

    const currentQuestion = questions[questionIndex];

    lblquestion.textContent = currentQuestion.question;

    lbla.textContent = currentQuestion.a;
    lblb.textContent = currentQuestion.b;
    lblc.textContent = currentQuestion.c;
    lbld.textContent = currentQuestion.d;

    rde.checked = true;

    startTimer();
}

Stage 13: Create getSelectedAnswer()

This function checks which radio button is selected. 

function getSelectedAnswer() {
    if (rde.checked) {
        return "";
    }

    if (rda.checked) {
        return "a";
    }

    if (rdb.checked) {
        return "b";
    }

    if (rdc.checked) {
        return "c";
    }

    if (rdd.checked) {
        return "d";
    }

    return "";
}

Stage 14: Add Check Answer Button

<button onclick="checkAnswer()">Check Answer</button>
function checkAnswer() {
    const selectedAnswer = getSelectedAnswer();

    if (selectedAnswer === "") {
        alert("Please select an answer.");
        return;
    }

    const correctAnswer = questions[questionIndex].answer;

    if (selectedAnswer === correctAnswer) {
        alert("Correct answer!");
        loadNextQuestion();
    } else {
        alert("Wrong answer.");
    }
}

Stage 15: Create endKbc()

function endKbc(message) {
    clearInterval(timerId);

    lblquestion.textContent = message;

    rdb.style.display = "none";
    rdc.style.display = "none";
    rdd.style.display = "none";

    lbltimer.textContent = "0";
}

Stage 19: Speak the Question and Options

We use the browser's speech feature: SpeechSynthesisUtterance. 

function speakQuestion() {
    if (questionIndex < 0 || questionIndex >= questions.length) {
        return;
    }

    const currentQuestion = questions[questionIndex];

    const text =
        "Question for " + rewards[questionIndex] + ". " +
        currentQuestion.question +
        ". Option A. " + currentQuestion.a +
        ". Option B. " + currentQuestion.b +
        ". Option C. " + currentQuestion.c +
        ". Option D. " + currentQuestion.d + ".";

    window.speechSynthesis.cancel();

    const speech = new SpeechSynthesisUtterance(text);
    speech.lang = "en-IN";
    speech.rate = 0.85;
    speech.pitch = 1;

    window.speechSynthesis.speak(speech);
}

Add a button to repeat the question:

<button onclick="speakQuestion()">Speak Again</button>
